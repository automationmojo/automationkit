[docs]    def answered_by(self, rec: 'DnsRecord') -> bool:
        """
            Returns true if the question is answered by the record.
        """
        bval = self.eclass == rec.eclass and \
            (self.etype == rec.etype or self.etype == DnsEntryType.ANY) and \
            self.name == rec.name
        return bval
