import time

FORMAT_DATETIME = "%Y-%m-%dT%H:%M:%S"


[docs]def format_time_with_fractional(tsecs):
    sec_comp = int(tsecs)
    frac_comp = tsecs - sec_comp
    dtstr = "%s.%03d" % (time.strftime(FORMAT_DATETIME), frac_comp)
    return dtstr
